# 构建一个es6项目教程

1.初始化

yarn init
yarn add webpack webpack-cli --save-dev

2.项目结构

3.代码结构

webpack.config.js:
var path = require('path');

module.exports = {
  mode: 'development',
  entry: './src/index.js',
  output: {
    path: path.resolve(__dirname, 'dist'),
    filename: 'main.js'
  }
};

package.json:
{
  "name": "es6test",
  "version": "1.0.1",
  "description": "es6测试",
  "main": "index.js",
  "author": "chen",
  "license": "MIT",
  "scripts": {
        "test": "echo \"Error: no test specified\" && exit 1",
        "build": "webpack"
      },
  "dependencies": {
    "webpack": "^4.41.0",
    "webpack-cli": "^3.3.9"
  }
}

index.html:
<!doctype html>
<html>
  <head>
    <title>起步</title>
  </head>
  <body>
    <script src="main.js"></script>
  </body>
</html>

index.js:
function component() {
  var element = document.createElement('div');
  element.innerHTML = 'hello webpack!';

  return element;
}

document.body.appendChild(component());
